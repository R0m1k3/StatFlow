# Étape 1 : Construire l'application React
FROM node:18-alpine AS builder
WORKDIR /app
COPY package.json .
# Note: package-lock.json n'est pas fourni, donc on utilise seulement package.json
RUN npm install
COPY . .
RUN npm run build

# Étape 2 : Servir l'application avec un serveur Node.js (Express)
FROM node:18-alpine
WORKDIR /app

# Copier uniquement les fichiers nécessaires pour la production
COPY package.json .
COPY server.js .

# Installer uniquement les dépendances de production
RUN npm install --omit=dev

# Copier les fichiers construits de l'étape 'builder'
COPY --from=builder /app/dist ./dist

# Exposer le port sur lequel le serveur écoute
EXPOSE 3000

# Commande pour démarrer le serveur
CMD ["npm", "start"]
